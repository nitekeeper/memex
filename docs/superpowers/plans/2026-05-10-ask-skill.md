def _parse_result(result):
    try:
        return json.loads(result.stdout)
    except json.JSONDecodeError as exc:
        raise AssertionError(
            f"search.py stdout was not valid JSON.\n"
            f"stdout: {result.stdout!r}\n"
            f"stderr: {result.stderr!r}"
        ) from exc


def _write_page(wiki_dir, slug, title, body, status="approved", tags=None):
    tags_yaml = ""
    if tags:
        tags_yaml = "tags:\n" + "".join(f"  - {t}\n" for t in tags)
    content = (
        f"---\n"
        f"id: test:wiki:{slug}\n"
        f"title: {title}\n"
        f"status: {status}\n"
        f"created: 2026-05-10\n"
        f"updated: 2026-05-10\n"
        f"{tags_yaml}"
        f"---\n\n"
        f"{body}\n"
    )
    (wiki_dir / f"{slug}.md").write_text(content, encoding="utf-8")


def _build_db(ai_dir):
    subprocess.run(
        [sys.executable, REBUILD_SCRIPT, str(ai_dir)],
        capture_output=True, text=True, check=True,
    )

```python
import argparse
import json
import os
import sqlite3
import sys
from typing import Optional


def _connect(db_path: str) -> sqlite3.Connection:
    conn = sqlite3.connect(db_path)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode = WAL")
    conn.execute("PRAGMA synchronous = NORMAL")
    return conn


def search(
    db_path: str,
    query: str,
    limit: int = 10,
    status: Optional[str] = None,
    tags: Optional[list] = None,
) -> dict:
    if not query.strip():
        raise ValueError("query cannot be empty")
    if not os.path.exists(db_path):
        raise FileNotFoundError(f"DB not found: {db_path}")

    conn = _connect(db_path)
    try:
        sql = """
            SELECT p.id, p.title, p.file_path, p.status, p.updated,
                   snippet(pages_fts, 2, '[', ']', '...', 20) AS snippet,
                   bm25(pages_fts) AS score
            FROM pages_fts
            JOIN pages p ON pages_fts.rowid = p.rowid
            WHERE pages_fts MATCH ?
        """
        params: list = [query]

        if status:
            sql += " AND p.status = ?"
            params.append(status)

        if tags:
            for tag in tags:
                sql += " AND p.id IN (SELECT page_id FROM page_tags WHERE tag = ?)"
                params.append(tag)

        sql += " ORDER BY bm25(pages_fts) LIMIT ?"
        params.append(limit)

        rows = conn.execute(sql, params).fetchall()
        return {
            "query": query,
            "results": [dict(row) for row in rows],
        }
    finally:
        conn.close()


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Search memex.db via FTS5.")
    parser.add_argument("ai_dir", help="Path to the project's .ai/ directory")
    parser.add_argument("query", help="FTS5 search string")
    parser.add_argument("--limit", type=int, default=10)
    parser.add_argument("--status", default=None)
    parser.add_argument("--tag", action="append", dest="tags", default=None)
    args = parser.parse_args()

    db_path = os.path.join(args.ai_dir, "memex.db")

    try:
        result = search(
            db_path, args.query,
            limit=args.limit,
            status=args.status,
            tags=args.tags or [],
        )
        print(json.dumps(result, indent=2))
    except (ValueError, FileNotFoundError) as e:
        print(str(e), file=sys.stderr)
        sys.exit(1)
```

# ask — tiered knowledge resolution

## Invocation

Triggered when:
- The user explicitly invokes `/ask <question>`
- The agent judges the project wiki may contain a relevant answer

## Project root detection

- **Zero detectable projects** (no `.ai/wiki/` found): stop. Tell user: "No project wiki found. Create an `.ai/wiki/` directory in your target project root before running ask."
- **Exactly one**: proceed. Announce: "Searching `<path>/.ai/`."
- **More than one**: ask the user which project. Wait for explicit choice.

If `memex.db` is absent at the confirmed root, note: "Wiki found but no DB index — skipping local search." Skip to Tier 2.

---

## Tier 1 — Local wiki

**Step 1.** Run from the confirmed project root:
```
python scripts/search.py <ai_dir> "<question>" --limit 10
```
On non-zero exit: show stderr; skip to Tier 2.

**Step 2.** Parse JSON. If `results` is empty, skip to Tier 2.

**Step 3.** Read results. Judge sufficiency — BM25 scores are not a threshold; read the content.
- **Sufficient**: present answer, cite wiki pages by title and path. Done.
- **Insufficient**: proceed to Tier 2.

---

## Tier 2 — Web search

**Step 1.** Run `WebSearch` with the original question.

**Step 2.** Synthesize and present an answer from the most relevant results. Cite sources (URLs).

**Step 3.** After presenting the answer, identify findings that are project-relevant and durable (design decisions, architecture patterns — not one-off fixes or ephemeral results).

**Step 4.** For each durable finding, offer: "I found relevant information on [topic] — want me to capture it to the wiki?" If yes, invoke the `capture` skill (normal approval gate).

If web search returns no useful results, proceed to Tier 3.

---

## Tier 3 — Model knowledge

**Step 1.** Answer from training knowledge.

**Step 2.** Append the mandatory disclosure block after the answer:

```
---
**Source:** Model knowledge
**Confidence:** <N>%
**Note:** This answer was not found in the project wiki or via web search. It is based on
training knowledge and may be incomplete, outdated, or incorrect. Verify before acting on it.
---
```

`<N>`: honest self-assessment 0–100. Always shown when Tier 3 is used — no exceptions.

---

## Error handling

| Situation | Action |
|---|---|
| `search.py` exits non-zero | Show stderr; skip to Tier 2. |
| `memex.db` absent | Note "no DB index — skipping local search"; skip to Tier 2. |
| Web search unavailable | Note "web search unavailable"; skip to Tier 3. |
| No detectable project | Stop: "No project wiki found. Create an `.ai/wiki/` directory in your target project root." |
```

```markdown
# ask — reference

## search.py CLI

```
python scripts/search.py <ai_dir> <query> [--limit N] [--status STATUS] [--tag TAG]
```

| Arg | Required | Notes |
|---|---|---|
| `ai_dir` | yes | Path to `.ai/` directory. DB resolved as `<ai_dir>/memex.db`. |
| `query` | yes | FTS5 search string. Quote if it contains spaces. |
| `--limit` | no | Max results. Default: 10. |
| `--status` | no | Filter: `draft`, `approved`, or `archived`. |
| `--tag` | no | Filter by tag. Repeatable: `--tag auth --tag session`. |

Exit 0 on success (empty results = exit 0). Non-zero on error. Errors go to stderr.

## Output format

```json
{
  "query": "token expiry",
  "results": [
    {
      "id": "myproject:wiki:auth-design",
      "title": "Auth design decisions",
      "file_path": ".ai/wiki/auth-design.md",
      "status": "approved",
      "updated": "2026-05-10",
      "snippet": "...handles [token] [expiry] by refreshing...",
      "score": -1.23
    }
  ]
}
```

`score`: BM25 value. Lower (more negative) = more relevant. Use for ordering only, not as a threshold.

## Disclosure block (Tier 3)

```
---
**Source:** Model knowledge
**Confidence:** <N>%
**Note:** This answer was not found in the project wiki or via web search. It is based on
training knowledge and may be incomplete, outdated, or incorrect. Verify before acting on it.
---
```

`<N>`: integer 0–100, honest self-assessment. This block is mandatory whenever Tier 3 is used.
```
